#include <WiFi.h>
#include <WebServer.h>

// =========================
// PINES
// =========================
#define UP_VISITA      19
#define DOWN_VISITA    14
#define RST_VISITA     23

// =========================
// WIFI
// =========================
const char* ssid = "TABLERO_VOLEY";
const char* password = "12345678";

WebServer server(80);

// =========================
// CONFIG
// =========================
const int PULSO_US = 50;
const int RESET_MS = 5;
const int BLOQUEO_MS = 180;   // evita doble toque muy seguido

unsigned long ultimoComando = 0;

// =========================
// FUNCIONES 74LS192
// =========================
bool bloqueado() {
  unsigned long ahora = millis();
  if (ahora - ultimoComando < BLOQUEO_MS) return true;
  ultimoComando = ahora;
  return false;
}

void estadoReposo() {
  digitalWrite(UP_VISITA, HIGH);
  digitalWrite(DOWN_VISITA, HIGH);
  digitalWrite(RST_VISITA, LOW);   // CLR inactivo
}

void sumaVisita() {
  digitalWrite(RST_VISITA, LOW);
  digitalWrite(DOWN_VISITA, HIGH);

  digitalWrite(UP_VISITA, LOW);
  delayMicroseconds(PULSO_US);
  digitalWrite(UP_VISITA, HIGH);
}

void restaVisita() {
  digitalWrite(RST_VISITA, LOW);
  digitalWrite(UP_VISITA, HIGH);

  digitalWrite(DOWN_VISITA, LOW);
  delayMicroseconds(PULSO_US);
  digitalWrite(DOWN_VISITA, HIGH);
}

void resetVisita() {
  digitalWrite(UP_VISITA, HIGH);
  digitalWrite(DOWN_VISITA, HIGH);

  digitalWrite(RST_VISITA, HIGH);
  delay(RESET_MS);
  digitalWrite(RST_VISITA, LOW);
}

// =========================
// RUTAS WEB
// =========================
void handleRoot() {
  server.send(200, "text/html", paginaHTML());
}

void handleUp() {
  if (!bloqueado()) {
    sumaVisita();
    Serial.println("SUMA VISITA");
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleDown() {
  if (!bloqueado()) {
    restaVisita();
    Serial.println("RESTA VISITA");
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

void handleReset() {
  if (!bloqueado()) {
    resetVisita();
    Serial.println("RESET VISITA");
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

// =========================
// SETUP
// =========================
void setup() {
  Serial.begin(115200);

  pinMode(UP_VISITA, OUTPUT);
  pinMode(DOWN_VISITA, OUTPUT);
  pinMode(RST_VISITA, OUTPUT);

  estadoReposo();

  WiFi.mode(WIFI_AP);
  WiFi.softAP(ssid, password);

  Serial.println();
  Serial.println("=== TABLERO VOLEY WIFI ===");
  Serial.print("Red: ");
  Serial.println(ssid);
  Serial.print("IP: ");
  Serial.println(WiFi.softAPIP());

  server.on("/", handleRoot);
  server.on("/up", handleUp);
  server.on("/down", handleDown);
  server.on("/reset", handleReset);

  server.begin();
  Serial.println("Servidor web iniciado");
}

// =========================
// LOOP
// =========================
void loop() {
  server.handleClient();
}
